Restaurant

/restaurant
methods allowed: ['GET', 'POST']
description: endpoint to view all restaurants or add a new restaurant.
get arguements: restaurant_id(optional)
post arguements: name, address, owner_name, contact_number

/restaurant/table
methods allowed: ['POST', 'GET', 'DELETE']
description: endpoint to get, delete or add table to restaurant.
get arguements: restaurant_id
post arguements: restaurant_id, table_capacity
delete arguements: restaurant_id, table_id

/restaurant/menu
methods allowed: ['POST', 'GET', 'DELETE']
description: endpoint to get restaurant menu or add new item to menu.
get arguements: restaurant_id
post arguements: restaurant_id, dish_name, cost, quantity("half", "full"), description, discount_percent.
delete arguements: restaurant_id, dish_id
